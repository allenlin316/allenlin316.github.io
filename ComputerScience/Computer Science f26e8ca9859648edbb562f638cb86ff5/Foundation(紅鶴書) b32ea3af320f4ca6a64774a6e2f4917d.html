#include <iostream>
using namespace std;
int main()
{
    // dynamically initialize 2D arrays 
    int **arr = new int *[5];
    for(int i=0; i<5; i++){
        arr[i] = new int[5];
    }
    //這樣後續arr操作就跟二維陣列一模一樣 (如 : arr[i][j] )
    
    // the following is clean up the pointer
    for(int i=0; i<5; i++) {
        delete [] arr[i];
    }
    delete [] arr;

}

C language

  • 二維動態陣列
#include <stdio.h>
#include <stdlib.h>

int main()
{
    int row = 3, col = 3;
    int input = 1;
    int **arr = (int**)malloc(row*sizeof(int*));
    for(int i=0; i<row; i++){
        arr[i] = (int*)malloc(col*sizeof(int));
    }
    for(int i=0; i<3; i++){
        for(int j=0; j<3; j++){
            arr[i][j] = input++;
        }
    }
    for(int i=0; i<3; i++){
        for(int j=0; j<3; j++){
            printf("%d ", arr[i][j]);
        }
        printf("\n");
    }

    //free the memory
    for(int i=0; i<row; i++){
        free(arr[i]);
    }
    free(arr);
}

Security

  • Key management
    • KDC (Key distribution center)
      • 在對稱式加密時使用KDC
      • To reduce the number of keys, each person establishes a shared secret key with the KDC
      • A secret key is established between the KDC and each member
    • Session keys
      • session key建立之前，A跟B透過自己與KDC做驗證
      • 要建立只有A、B兩方的秘密連線，要透過session key
      • session symmetric key 連線結束就沒用了
    • Certification authority (CA)
      • 在非對稱式加密中，想要公布 公鑰 給別人，但不想被竄改時使用
      • 承擔公鑰體系中公鑰的合法性檢驗的責任
      • CA把A方的公鑰與CA自己的私鑰加密
      • 到時候就用CA的公鑰解密拿出A的公鑰來用
